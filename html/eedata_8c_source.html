262UNS8 EEPROM_commit(){
263 FLASH_EraseInitTypeDef eraseInitStruct;
264 uint32_t PageError = 0;
265
266 if (page == NULL)
267 return 0;
268
269 __disable_irq();
270 HAL_FLASH_Unlock();
271
272 //Erase the page
273 eraseInitStruct.TypeErase = TYPEERASE_PAGES; // Just erase pages
274 eraseInitStruct.Page = 127 - EEPROM_PAGE_SIZE; // Erase the EEPROM pages, leaving parameters untouched
275 eraseInitStruct.NbPages = EEPROM_PAGE_SIZE;
276
277 if (HAL_FLASHEx_Erase(&eraseInitStruct, &PageError) != HAL_OK) {
278 HAL_FLASH_Lock(); // Re-lock the flash afterwards
279 free(page);
280 page = NULL;
281 __enable_irq();
282 return 0;
283 }
284
285 //Rewrite the page in flash
286 for (int i = 0; i < EEPROM_PAGE_SIZE * FLASH_PAGE_SIZE; i += 8) // Program into flash in double word increments (8 bytes)
287 {
288 if (HAL_FLASH_Program(TYPEPROGRAM_DOUBLEWORD, EEPROM_START_ADDRESS + i, *(uint64_t *)(page + i)) != HAL_OK)
289 {
290 return 0;
291 }
292 }
293
294 HAL_FLASH_Lock(); // Re-lock the flash afterwards
295 __enable_irq();
296
297 free(page);
298 page = NULL;
299 return 1;
300}
